# Midterm
## рЃЎрЃЮрЃарЃћрЃџрЃљрЃфрЃўрЃўрЃА рЃЏрЃЮрЃФрЃћрЃЉрЃюрЃљ
### рЃюрЃљрЃЉрЃўрЃ»рЃў 1
рЃљрЃЏрЃЮрЃўрЃгрЃћрЃарЃљ рЃДрЃЋрЃћрЃџрЃљ рЃгрЃћрЃарЃбрЃўрЃџрЃўрЃА x рЃЊрЃљ y рЃЎрЃЮрЃЮрЃарЃЊрЃўрЃюрЃљрЃбрЃћрЃЉрЃў:

x = [-9.5, -6.2, -5, -1.5, 1, 3, 5, 7]

y = [1, 2, -1, 0.1, -2, 0, -3, -2]

### рЃюрЃљрЃЉрЃўрЃ»рЃў 2
рЃњрЃљрЃЏрЃЮрЃўрЃЌрЃЋрЃљрЃџрЃљ рЃАрЃљрЃерЃБрЃљрЃџрЃЮ рЃЏрЃюрЃўрЃерЃЋрЃюрЃћрЃџрЃЮрЃЉрЃћрЃЉрЃў x рЃЊрЃљ y рЃЏрЃЮрЃюрЃљрЃфрЃћрЃЏрЃћрЃЉрЃўрЃАрЃЌрЃЋрЃўрЃА:

x рЃАрЃљрЃе = -0.775, y рЃАрЃљрЃе = -0.6125

### рЃюрЃљрЃЉрЃўрЃ»рЃў 3
рЃўрЃЏрЃўрЃА рЃњрЃљрЃЏрЃЮ, рЃарЃЮрЃЏ рЃЏрЃЮрЃюрЃљрЃфрЃћрЃЏрЃЌрЃљ рЃфрЃљрЃџ-рЃфрЃљрЃџрЃЎрЃћ рЃЊрЃљрЃЏрЃБрЃерЃљрЃЋрЃћрЃЉрЃљрЃА рЃФрЃљрЃџрЃўрЃљрЃю рЃЊрЃўрЃЊрЃў рЃЊрЃарЃЮ рЃЊрЃљрЃАрЃГрЃўрЃарЃЊрЃћрЃЉрЃљ, рЃЏрЃўрЃќрЃљрЃюрЃерЃћрЃгрЃЮрЃюрЃўрЃџрЃўрЃљ рЃърЃўрЃЌрЃЮрЃюрЃўрЃА рЃЉрЃўрЃЉрЃџрЃўрЃЮрЃЌрЃћрЃЎрЃћрЃЉрЃўрЃА рЃњрЃљрЃЏрЃЮрЃДрЃћрЃюрЃћрЃЉрЃљ рЃерЃћрЃЏрЃЊрЃћрЃњрЃў рЃюрЃљрЃЉрЃўрЃ»рЃћрЃЉрЃўрЃА рЃерЃћрЃАрЃљрЃАрЃарЃБрЃџрЃћрЃЉрЃџрЃљрЃЊ:

        import numpy as np
        x = np.array([-9.5, -6.2, -5, -1.5, 1, 3, 5, 7])
        y = np.array([1, 2, -1, 0.1, -2, 0, -3, -2])
        x_mean = np.mean(x)
        y_mean = np.mean(y)

рЃАрЃ«рЃЋрЃљрЃЮрЃЉрЃћрЃЉрЃўрЃА рЃњрЃљрЃЏрЃЮрЃЌрЃЋрЃџрЃљ:

        x_diff = x - x_mean
        y_diff = y - y_mean

### рЃюрЃљрЃЉрЃўрЃ»рЃў 4
рЃЏрЃарЃўрЃфрЃ«рЃЋрЃћрЃџрЃўрЃА рЃЊрЃљ рЃЏрЃюрЃўрЃерЃЋрЃюрЃћрЃџрЃўрЃА рЃюрЃљрЃЏрЃарЃљрЃЋрЃџрЃћрЃЉрЃўрЃА рЃњрЃљрЃЏрЃЮрЃЌрЃЋрЃџрЃљ:

        xy_diff_product = x_diff * y_diff
        x_diff_sq = x_diff ** 2
        y_diff_sq = y_diff ** 2

### рЃюрЃљрЃЉрЃўрЃ»рЃў 5
рЃюрЃљрЃЏрЃарЃљрЃЋрЃџрЃћрЃЉрЃўрЃА рЃерЃћрЃ»рЃљрЃЏрЃћрЃЉрЃљ рЃЏрЃарЃўрЃфрЃ«рЃЋрЃћрЃџрЃерЃў:

        numerator = np.sum(xy_diff_product)

### рЃюрЃљрЃЉрЃўрЃ»рЃў 6
рЃюрЃљрЃЏрЃарЃљрЃЋрЃџрЃћрЃЉрЃўрЃА рЃерЃћрЃ»рЃљрЃЏрЃћрЃЉрЃљ, рЃњрЃљрЃЏрЃарЃљрЃЋрЃџрЃћрЃЉрЃљ рЃЊрЃљ рЃцрЃћрЃАрЃЋрЃўрЃА рЃљрЃЏрЃЮрЃДрЃЋрЃљрЃюрЃљ рЃЏрЃюрЃўрЃерЃЋрЃюрЃћрЃџрЃерЃў:

        denominator = np.sqrt(np.sum(x_diff_sq)) * np.sqrt(np.sum(y_diff_sq))

### рЃюрЃљрЃЉрЃўрЃ»рЃў 7
рЃЏрЃўрЃдрЃћрЃЉрЃБрЃџрЃў рЃЏрЃюрЃўрЃерЃЋрЃюрЃћрЃџрЃЮрЃЉрЃћрЃЉрЃўрЃА рЃћрЃарЃЌрЃЏрЃљрЃюрЃћрЃЌрЃќрЃћ рЃњрЃљрЃДрЃЮрЃцрЃљ:

        r = numerator / denominator

### рЃерЃћрЃЊрЃћрЃњрЃў
рЃърЃўрЃарЃАрЃЮрЃюрЃўрЃА рЃЎрЃЮрЃарЃћрЃџрЃљрЃфрЃўрЃўрЃА рЃЎрЃЮрЃћрЃцрЃўрЃфрЃўрЃћрЃюрЃбрЃў -0.7558910476323901-рЃўрЃА рЃбрЃЮрЃџрЃўрЃљ, рЃарЃљрЃф рЃЮрЃа рЃфрЃЋрЃџрЃљрЃЊрЃА рЃерЃЮрЃарЃўрЃА (x рЃЊрЃљ y) рЃАрЃљрЃЎрЃЏрЃљрЃЮрЃЊ рЃФрЃџрЃўрЃћрЃа рЃБрЃЎрЃБрЃЊрЃљрЃЏрЃЮрЃЎрЃўрЃЊрЃћрЃЉрЃБрЃџрЃћрЃЉрЃљрЃА рЃљрЃЕрЃЋрЃћрЃюрЃћрЃЉрЃА. рЃБрЃЎрЃћрЃЌрЃћрЃАрЃў рЃњрЃљрЃњрЃћрЃЉрЃўрЃАрЃЌрЃЋрЃўрЃА рЃљрЃЋрЃљрЃњрЃЮрЃЌ рЃњрЃљрЃЉрЃюрЃћрЃЋрЃўрЃА рЃњрЃарЃљрЃцрЃўрЃЎрЃў:

        import matplotlib.pyplot as plt
        plt.figure(figsize=(7, 5))
        plt.scatter(x, y, color='blue', edgecolor='black')
        plt.title('x рЃЊрЃљ y рЃЎрЃЮрЃЮрЃарЃЊрЃўрЃюрЃљрЃбрЃћрЃЉрЃўрЃА рЃњрЃљрЃЉрЃюрЃћрЃЋрЃўрЃА рЃњрЃарЃљрЃцрЃўрЃЎрЃў')
        plt.xlabel('x')
        plt.ylabel('y')
        plt.grid(True)
        plt.tight_layout()

рЃњрЃарЃљрЃцрЃўрЃЎрЃќрЃћ рЃЕрЃљрЃюрЃА, рЃарЃЮрЃЏ y-рЃЏрЃюрЃўрЃерЃЋрЃюрЃћрЃџрЃЮрЃЉрЃћрЃЉрЃўрЃА рЃЎрЃџрЃћрЃЉрЃўрЃАрЃљрЃА, x-рЃЏрЃюрЃўрЃерЃЋрЃюрЃћрЃџрЃЮрЃЉрЃћрЃЉрЃў рЃўрЃќрЃарЃЊрЃћрЃЉрЃљ, рЃарЃљрЃф рЃљрЃЊрЃљрЃАрЃбрЃБрЃарЃћрЃЉрЃА рЃЏрЃўрЃдрЃћрЃЉрЃБрЃџ рЃЎрЃЮрЃћрЃцрЃўрЃфрЃўрЃћрЃюрЃбрЃА.




## рЃАрЃърЃљрЃЏ рЃћрЃџ-рЃцрЃЮрЃАрЃбрЃўрЃА рЃњрЃљрЃЏрЃЮрЃЋрЃџрЃћрЃюрЃљ
### 1. ![рЃЏрЃЮрЃфрЃћрЃЏрЃБрЃџрЃў рЃЊрЃљрЃЋрЃљрЃџрЃћрЃЉрЃљ рЃерЃћрЃАрЃарЃБрЃџрЃћрЃЉрЃБрЃџрЃўрЃљ рЃерЃћрЃЏрЃЊрЃћрЃњ рЃЏрЃЮрЃюрЃљрЃфрЃћрЃЏрЃЌрЃљ рЃюрЃљрЃЎрЃарЃћрЃЉрЃќрЃћ] https://github.com/Nodar-Melkonyan/midterm/blob/main/nodar_melkonyan_1_81429765.csv

### 2. ![рЃџрЃЮрЃњрЃўрЃАрЃбрЃўрЃЎрЃБрЃарЃў рЃарЃћрЃњрЃарЃћрЃАрЃўрЃўрЃА рЃърЃарЃЮрЃњрЃарЃљрЃЏрЃљ] https://github.com/Nodar-Melkonyan/midterm/blob/main/Logistic_Regression.py

рЃЏрЃЮрЃюрЃљрЃфрЃћрЃЏрЃћрЃЉрЃўрЃА рЃгрЃљрЃўрЃЎрЃўрЃЌрЃ«рЃљ pandas рЃЉрЃўрЃЉрЃџрЃўрЃЮрЃЌрЃћрЃЎрЃўрЃА read_csv() рЃцрЃБрЃюрЃЦрЃфрЃўрЃўрЃА рЃЏрЃћрЃерЃЋрЃћрЃЮрЃЉрЃўрЃЌ рЃерЃћрЃАрЃљрЃЉрЃљрЃЏрЃўрЃА рЃЊрЃўрЃарЃћрЃЦрЃбрЃЮрЃарЃўрЃљрЃерЃў рЃЏрЃЮрЃЌрЃљрЃЋрЃАрЃћрЃЉрЃБрЃџрЃў рЃцрЃљрЃўрЃџрЃўрЃЊрЃљрЃю:

        df = pd.read_csv("nodar_melkonyan_1_81429765.csv")

рЃерЃћрЃЏрЃЊрЃћрЃњ, рЃЏрЃЮрЃюрЃљрЃфрЃћрЃЏрЃћрЃЉрЃў рЃЊрЃљрЃўрЃДрЃЮ рЃЮрЃарЃљрЃЊ: рЃърЃљрЃарЃљрЃЏрЃћрЃбрЃарЃћрЃЉрЃљрЃЊ, рЃарЃЮрЃЏрЃџрЃћрЃЉрЃАрЃљрЃф рЃерЃћрЃБрЃФрЃџрЃўрЃљрЃЌ рЃќрЃћрЃњрЃљрЃЋрЃџрЃћрЃюрЃљ рЃ░рЃЦрЃЮрЃюрЃЊрЃћрЃЌ рЃАрЃљрЃЏрЃўрЃќрЃюрЃћрЃќрЃћ: рЃљрЃарЃўрЃА рЃЌрЃБ рЃљрЃарЃљ рЃАрЃърЃљрЃЏрЃў рЃерЃћрЃАрЃљрЃЉрЃљрЃЏрЃўрЃАрЃў рЃгрЃћрЃарЃўрЃџрЃў:

        X = df[['words', 'links', 'capital_words', 'spam_word_count']]
        y = df['is_spam']

рЃЏрЃЮрЃюрЃљрЃфрЃћрЃЏрЃћрЃЉрЃўрЃА рЃЏрЃЮрЃЏрЃЊрЃћрЃЋрЃюрЃЮ рЃЊрЃљрЃДрЃЮрЃцрЃљ рЃњрЃљрЃАрЃљрЃгрЃЋрЃарЃЌрЃю рЃЊрЃљ рЃАрЃљрЃбрЃћрЃАрЃбрЃЮ рЃюрЃљрЃЎрЃарЃћрЃЉрЃћрЃЉрЃљрЃЊ рЃерЃћрЃЏрЃЊрЃћрЃњрЃў рЃЊрЃљрЃЏрЃБрЃерЃљрЃЋрЃћрЃЉрЃўрЃАрЃЌрЃЋрЃўрЃА, рЃАрЃљрЃЊрЃљрЃф рЃњрЃљрЃАрЃљрЃгрЃЋрЃарЃЌрЃюрЃў рЃЏрЃЮрЃюрЃљрЃфрЃћрЃЏрЃћрЃЉрЃўрЃА рЃгрЃўрЃџрЃў 70% рЃљрЃарЃўрЃА:

        X_train, X_test, y_train, y_test = train_test_split(X, y, train_size=0.7, random_state=42)

рЃерЃћрЃАрЃљрЃЉрЃљрЃЏрЃўрЃАрЃў рЃЏрЃЮрЃЊрЃћрЃџрЃў рЃЊрЃљ рЃњрЃљрЃгрЃЋрЃарЃЌрЃюрЃљ:

        model = LogisticRegression()
        model.fit(X_train, y_train)

рЃЎрЃЮрЃћрЃцрЃўрЃфрЃўрЃћрЃюрЃбрЃћрЃЉрЃўрЃА рЃњрЃљрЃЏрЃЮрЃЌрЃЋрЃџрЃљ:

        coefficients = pd.Series(model.coef_[0], index=X.columns)
        print (coefficients)

рЃерЃћрЃЊрЃћрЃњрЃћрЃЉрЃў:

words              0.006916
links              0.915510
capital_words      0.517343

рЃЕрЃљрЃюрЃА, рЃарЃЮрЃЏ рЃАрЃўрЃбрЃДрЃЋрЃћрЃЉрЃўрЃА рЃарЃљрЃЮрЃЊрЃћрЃюрЃЮрЃЉрЃљрЃА рЃљрЃа рЃљрЃЦрЃЋрЃА рЃарЃљрЃўрЃЏрЃћ рЃљрЃарЃАрЃћрЃЉрЃўрЃЌрЃў рЃњрЃљрЃЋрЃџрЃћрЃюрЃљ рЃАрЃърЃљрЃЏ-рЃгрЃћрЃарЃўрЃџрЃћрЃЉрЃўрЃА рЃљрЃЏрЃЮрЃфрЃюрЃЮрЃЉрЃљрЃерЃў. рЃљрЃЏрЃљрЃЋрЃЊрЃарЃЮрЃБрЃџрЃљрЃЊ, рЃЌрЃўрЃЌрЃЦрЃЏрЃўрЃА рЃЉрЃЏрЃБрЃџрЃћрЃЉрЃўрЃА рЃарЃљрЃЮрЃЊрЃћрЃюрЃЮрЃЉрЃљ рЃБрЃЏрЃюрЃўрЃерЃЋрЃюрЃћрЃџрЃЮрЃЋрЃљрЃюрЃћрЃАрЃў рЃцрЃљрЃЦрЃбрЃЮрЃарЃўрЃљ рЃљрЃАрЃћрЃЌрЃў рЃгрЃћрЃарЃўрЃџрЃћрЃЉрЃўрЃА рЃњрЃљрЃЏрЃЮрЃАрЃљрЃЋрЃџрЃћрЃюрЃљрЃЊ. рЃњрЃљрЃЏрЃЮрЃўрЃарЃЕрЃћрЃЋрЃљ, рЃљрЃАрЃћрЃЋрЃћ, рЃАрЃърЃљрЃЏ-рЃАрЃўрЃбрЃДрЃЋрЃћрЃЉрЃў, рЃЌрЃБрЃЏрЃфрЃљ рЃљрЃарЃљ рЃўрЃАрЃћрЃЌрЃў "рЃАрЃўрЃФрЃџрЃўрЃћрЃарЃўрЃЌ", рЃарЃЮрЃњрЃЮрЃарЃф рЃЉрЃЏрЃБрЃџрЃћрЃЉрЃў. рЃЊрЃўрЃЊрЃў рЃљрЃАрЃЮрЃћрЃЉрЃўрЃА рЃерЃћрЃЏрЃфрЃЋрЃћрЃџ рЃАрЃўрЃбрЃДрЃЋрЃћрЃЉрЃќрЃћ рЃЎрЃў рЃфрЃљрЃџрЃАрЃљрЃ«рЃљ рЃърЃљрЃАрЃБрЃ«рЃА рЃЋрЃћрЃа рЃњрЃљрЃЋрЃфрЃћрЃЏрЃЌ: 0.517343 рЃљрЃа рЃўрЃФрЃџрЃћрЃЋрЃљ рЃАрЃљрЃЎрЃЏрЃљрЃарЃўрЃА рЃўрЃюрЃцрЃЮрЃарЃЏрЃљрЃфрЃўрЃљрЃА рЃљрЃЏ рЃърЃљрЃарЃљрЃЏрЃћрЃбрЃарЃўрЃА рЃњрЃљрЃЋрЃџрЃћрЃюрЃљрЃќрЃћ.
